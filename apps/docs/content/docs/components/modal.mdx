A centered modal dialog overlay. Defined inside screens and triggered by `:to "#id"`.

## Usage

```wire preview:400
(wire
  (screen main "Main" :desktop
    (box :col :center :full
      (button "Open Modal" :primary :to "#my-modal"))

    (modal :id "my-modal"
      (box :col :gap 16 :padding 24
        (text "Modal Content" :high)
        (text "This is a modal dialog.")
        (button "Close" :ghost :to :close)))))
```

## Opening a Modal

```wire
(button "Open Modal" :to "#my-modal")
```

## Closing a Modal

```wire
(button "Close" :to :close)
```

## Examples

### Initially Open Modal

```wire preview:400
(wire
  (screen main "Main" :desktop
    (box :col :center :full :padding 24
      (text "Content behind modal" :low))

    (modal :id "welcome" :open
      (box :col :gap 16 :padding 24
        (text "Welcome!" :high)
        (text "This modal opens automatically.")
        (button "Got it" :primary :to :close)))))
```

### Confirmation Modal

```wire preview:400
(wire
  (screen main "Main" :desktop
    (box :col :center :full :padding 24
      (button "Delete Item" :danger :to "#confirm"))

    (modal :id "confirm"
      (box :col :gap 16 :padding 24
        (text "Delete Item?" :high)
        (text "This action cannot be undone." :low)
        (box :row :end :gap 12
          (button "Cancel" :ghost :to :close)
          (button "Delete" :danger))))))
```

### Form Modal

```wire preview:500
(wire
  (screen main "Main" :desktop
    (box :col :center :full :padding 24
      (button "Add User" :primary :to "#add-user"))

    (modal :id "add-user"
      (box :col :gap 20 :padding 24
        (text "Add New User" :high)
        (form :col :gap 16
          (input "Name")
          (input "Email" :type email)
          (input "Role" :type select :options "Admin,Editor,Viewer"))
        (box :row :end :gap 12
          (button "Cancel" :ghost :to :close)
          (button "Add User" :primary))))))
```

### Info Modal

```wire preview:400
(wire
  (screen main "Main" :desktop
    (box :col :center :full :padding 24
      (icon "info" :to "#info"))

    (modal :id "info"
      (box :col :gap 16 :padding 24
        (box :row :between :center
          (text "About" :high)
          (icon "x" :to :close))
        (text "This application helps you create wireframes using a simple text-based syntax.")
        (button "Got it" :primary :full :to :close)))))
```

### Image Preview Modal

```wire preview:500
(wire
  (screen gallery "Gallery" :desktop
    (box :padding 24
      (box :grid :cols 3 :gap 8
        (image "Photo 1" :to "#preview")
        (image "Photo 2" :to "#preview")
        (image "Photo 3" :to "#preview")))

    (modal :id "preview"
      (box :col :gap 16 :padding 24
        (box :row :end
          (icon "x" :to :close))
        (image "Preview" :width 600 :height 400)))))
```

### Multi-step Modal

```wire preview:500
(wire
  (screen main "Main" :desktop
    (box :col :center :full :padding 24
      (button "Start Wizard" :primary :to "#wizard"))

    (modal :id "wizard"
      (box :col :gap 20 :padding 24
        (progress :value 50)
        (text "Step 2 of 4" :low)
        (text "Choose Your Plan" :high)
        (box :col :gap 12
          (card :row :between :center :padding 16
            (text "Basic")
            (text "$9/mo"))
          (card :row :between :center :padding 16
            (text "Pro")
            (text "$29/mo")))
        (box :row :between
          (button "Back" :ghost)
          (button "Next" :primary))))))
```
